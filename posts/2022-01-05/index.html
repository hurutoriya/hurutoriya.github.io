import requests

# NOTE: put API KEY
API_KEY:str = ''

target_file:str = ""

with open(target_file) as f:
    txt = f.read()

params = {
            "auth_key": API_KEY,
            "text": txt,
            "source_lang": 'EN',
            "target_lang": 'JA'
        }

request = requests.post("https://api-free.deepl.com/v2/translate", data=params)

result = request.json()

with open("JA_"+target_file, "w") as text_file:
    text_file.write(result["translations"][0]["text"])
